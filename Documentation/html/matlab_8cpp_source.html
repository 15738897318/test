31 
32  int nSegment = (int)segment.size();
33  for (int i = 1; i < nSegment - 1; ++i) {
34  if ((segment[i] - segment[i-1] > threshold) &&
35  (segment[i] - segment[i+1] >= threshold))
36  {
37  peak_list.push_back(pair<double,int> (-segment[i], i));
38  }
39  }
40 
41  if (peak_list.empty())
42  return;
43 
44  // Code to sort the peaks by position. The first & last peaks should be such that between
45  // the peaks and the start / end of the segment there must be no 'straight line'
46  int nPeaks = (int)peak_list.size();
47  int n = peak_list[nPeaks - 1].second;
48  double minValue = segment[n], maxValue = segment[n];
49  for (int i = n+1; i < nSegment; ++i) {
50  minValue = min(minValue, segment[i]);
51  maxValue = max(maxValue, segment[i]);
52  }
53  if (maxValue == minValue)
54  peak_list.pop_back();
55 
56 
57  sort(peak_list.begin(), peak_list.end());
58  for (int i = 0; i < nPeaks; ++i){
59  int pos=peak_list[i].second;
60  if(pos==-1) continue;
61  for (int j = 0; j < nPeaks; ++j)
62  if(j!=i && peak_list[j].second!=-1 && abs(peak_list[j].second-pos) <= minPeakDistance)
63  peak_list[j].second=-1;
64  }
65 
66  for (int i = 0; i < nPeaks; ++i)
67  if(peak_list[i].second!=-1){
68  max_peak_locs.push_back(peak_list[i].second);

87  vector<double> corr_linear(vector<double> signal, vector<double> kernel, bool subtractMean) {
88  int m = (int)signal.size(), n = (int)kernel.size();
89 
90  // -meanValue
91  if (subtractMean) {
92  double meanValue = mean(signal);
93  for (int i = 0; i < m; ++i) signal[i] -= meanValue;
94  meanValue = mean(kernel);
95  for (int i = 0; i < n; ++i) kernel[i] -= meanValue;
96  }
97 
98  // padding of zeors
99  for(int i = m; i < m+n-1; i++) signal.push_back(0);
100  for(int i = n; i < m+n-1; i++) kernel.push_back(0);
101 
102  /* convolution operation */
103  vector<double> ans;
104  for(int i = 0; i < m+n-1; i++)
105  {
106  ans.push_back(0);
107  for(int j = 0; j <= i; j++)
108  ans[i] += signal[j]*kernel[i-j];
109  }
110 
111  for (int i = 0; i < n-1; ++i)
112  ans.pop_back();
113 
114  if (subtractMean) {
115  double minValue = *min_element(ans.begin(), ans.end());
116  if (minValue < 0)
117  for (int i = 0, sz = (int)ans.size(); i < sz; ++i)
118  ans[i] -= minValue;
119  }
120 
121  return ans;
122  }

176  int int_idx = (int) (idx+1e-9);

183  }else{
